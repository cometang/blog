## 数据绑定

```js
<template>
  <div id="app">
      {{a}}-----{{b[0]}}
  </div>
</template>

<script>
export default {    
  name: 'App',
  data(){  //绑定数据
    return{
      a:'1111',
      b:[1,2,3,4]
    }
  },
  methods:{   //方法

  },
  //钩子函数

  //计算属性

}
</script>

<style>
#app {
  font-family: Avenir, Helvetica, Arial, sans-serif;
  -webkit-font-smoothing: antialiased;
  -moz-osx-font-smoothing: grayscale;
  text-align: center;
  color: #2c3e50;
  margin-top: 60px;
}
</style>


```

### 方法的使用

```js
<template>
  <div id="app">
    {{ a }}-----{{ b[0] }}
    <p>
      {{ c }}
    </p>
    <button @click="add()">C+1</button>
  </div>
</template>

<script>
export default {
  name: "App",
  data() {
    //绑定数据
    return {
      a: "1111",
      b: [1, 2, 3, 4],
      c: 100,
    };
  },
  methods: {
    //方法
    add() {
      // this=== Vue
      this.c++;
    },
  },
  //钩子函数
	
  //计算属性
};
</script>

<style>
#app {
  font-family: Avenir, Helvetica, Arial, sans-serif;
  -webkit-font-smoothing: antialiased;
  -moz-osx-font-smoothing: grayscale;
  text-align: center;
  color: #2c3e50;
  margin-top: 60px;
}
</style>

```
